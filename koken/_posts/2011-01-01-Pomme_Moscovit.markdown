recipe:
  ingredients:
    - "krielaardappeltjes"
    - "zure room"
    - "lompviseitjes"
  instructions:
    - "Kook de aardappeltjes in licht gezouten water. Laat ze afkoelen."

    - "Lepel er een klein beetje zure room op en werk af met wat lompviseitjes."
